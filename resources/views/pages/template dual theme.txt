/* Dark particles */

const darkPts = Array.from({length:DARK_N}, () => ({
  x: Math.random()*innerWidth, y: Math.random()*innerHeight,
  vx:(Math.random()-0.5)*0.35, vy:(Math.random()-0.5)*0.35,
  r: Math.random()*2+0.8, op: Math.random()*0.45+0.1
}));

const lightPts = Array.from({length:LIGHT_N}, () => ({
  x: Math.random()*innerWidth, y: Math.random()*innerHeight,
  vx:(Math.random()-0.5)*0.38, vy:(Math.random()-0.5)*0.38,
  r: Math.random()*2.2+0.8, op: Math.random()*0.4+0.1
}));

function renderLoop() {
  ctx.clearRect(0, 0, canvas.width, canvas.height);
  if (theme === 'dark') renderDark();
  else renderLight();
  requestAnimationFrame(renderLoop);
}

function renderDark() {
  const CR=56, CG=139, CB=253;
  for (let i=0;i<DARK_N;i++) {
    const p=darkPts[i];
    p.x+=p.vx; p.y+=p.vy;
    if(p.x<0||p.x>canvas.width)  p.vx*=-1;
    if(p.y<0||p.y>canvas.height) p.vy*=-1;
    ctx.beginPath(); ctx.arc(p.x,p.y,p.r,0,Math.PI*2);
    ctx.fillStyle=`rgba(${CR},${CG},${CB},${p.op})`; ctx.fill();
    for (let j=i+1;j<DARK_N;j++) {
      const q=darkPts[j];
      const d=Math.hypot(p.x-q.x,p.y-q.y);
      if(d<DARK_MAX) {
        const a=(1-d/DARK_MAX)*0.15;
        ctx.beginPath(); ctx.moveTo(p.x,p.y); ctx.lineTo(q.x,q.y);
        ctx.strokeStyle=`rgba(${CR},${CG},${CB},${a})`; ctx.lineWidth=0.5; ctx.stroke();
      }
    }
    const md=Math.hypot(p.x-mx,p.y-my);
    if(md<120) {
      const a=(1-md/120)*0.35;
      ctx.beginPath(); ctx.moveTo(p.x,p.y); ctx.lineTo(mx,my);
      ctx.strokeStyle=`rgba(57,208,216,${a})`; ctx.lineWidth=0.9; ctx.stroke();
    }
  }
}

function renderLight() {
  const CR=37, CG=99, CB=235;
  for (let i=0;i<LIGHT_N;i++) {
    const p=lightPts[i];
    p.x+=p.vx; p.y+=p.vy;
    if(p.x<0||p.x>canvas.width)  p.vx*=-1;
    if(p.y<0||p.y>canvas.height) p.vy*=-1;
    ctx.beginPath(); ctx.arc(p.x,p.y,p.r,0,Math.PI*2);
    ctx.fillStyle=`rgba(${CR},${CG},${CB},${p.op})`; ctx.fill();
    for (let j=i+1;j<LIGHT_N;j++) {
      const q=lightPts[j];
      const d=Math.hypot(p.x-q.x,p.y-q.y);
      if(d<LIGHT_MAX) {
        const a=(1-d/LIGHT_MAX)*0.13;
        ctx.beginPath(); ctx.moveTo(p.x,p.y); ctx.lineTo(q.x,q.y);
        ctx.strokeStyle=`rgba(${CR},${CG},${CB},${a})`; ctx.lineWidth=0.5; ctx.stroke();
      }
    }
    const md=Math.hypot(p.x-mx,p.y-my);
    if(md<110) {
      const a=(1-md/110)*0.28;
      ctx.beginPath(); ctx.moveTo(p.x,p.y); ctx.lineTo(mx,my);
      ctx.strokeStyle=`rgba(${CR},${CG},${CB},${a})`; ctx.lineWidth=0.8; ctx.stroke();
    }
  }
}
